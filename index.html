import pandas as pd
import boto3
from datetime import datetime

def ingest_market_data(provider: str, symbols: list) -> dict:
    raw = fetch_from_provider(provider, symbols)

    df = pd.DataFrame(raw)
    df.dropna(subset=["close", "volume"], inplace=True)
    df["timestamp"] = pd.to_datetime(df["timestamp"])
    df.sort_values("timestamp", inplace=True)

    # Upload to S3 data lake
    s3_key = f"market/{provider}/{datetime.today().date()}.parquet"
    df.to_parquet(f"s3://data-lake/{s3_key}")

    return {"rows_ingested": len(df),
            "provider": provider,
            "status": "success"}
